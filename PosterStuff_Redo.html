# Clean column names
df.columns = df.columns.str.strip().str.replace(" ", "_")

# Convert dates
df['Date_of_Admission'] = pd.to_datetime(df['Date_of_Admission'], errors='coerce')
df['Discharge_Date'] = pd.to_datetime(df['Discharge_Date'], errors='coerce')

# Calculate length of stay
df['Length_of_Stay'] = (df['Discharge_Date'] - df['Date_of_Admission']).dt.days

# Drop non-predictive columns
df_model = df.drop(columns=['Name', 'Doctor', 'Hospital', 'Date_of_Admission', 'Discharge_Date', 'Room_Number'])

# Encode categorical variables
df_model = pd.get_dummies(df_model, drop_first=True)

# Check for target column
target_col = [col for col in df_model.columns if 'Test' in col and 'Result' in col]
if target_col:
    df_model = df_model.dropna(subset=target_col)
    df_model[target_col[0]] = df_model[target_col[0]].astype('category').cat.codes
else:
    raise KeyError("Target column 'Test_Results' not found. Check column names.")

# Train models
models = {
    "Dummy": DummyClassifier(strategy="most_frequent"),
    "Logistic Regression": LogisticRegression(max_iter=1000),
    "Decision Tree": DecisionTreeClassifier(max_depth=5),
    "Random Forest": RandomForestClassifier(n_estimators=100)
}

results = {}
for name, model in models.items():
    model.fit(X_train_scaled, y_train)
    y_pred = model.predict(X_test_scaled)
    results[name] = {
        "Accuracy": accuracy_score(y_test, y_pred),
        "Report": classification_report(y_test, y_pred, output_dict=True, zero_division=0)
    }
results_df = pd.DataFrame(results).T
print(results_df)

## Discussion and Impact

# **Predictive Power**: Random Forest outperformed other models, suggesting non-linear relationships in the data.
# **Key Drivers**: Age, Medical Condition, and Insurance Provider were top predictors of test outcomes.
# **Cost Trends**: Patients with positive test results had higher billing and longer stays, indicating greater resource use.
# **Hospital Planning**: These insights can guide staffing, bed allocation, and insurance negotiations.

## This analysis supports data-driven decision-making in healthcare operations and patient care optimization.

# Operational Impact:
# Having the power to test outcomes in advance will give hospitals the ability to correctly allocate their resources.
# An example of this would be if the model identifies patients who have a high probability of developing abnormal test results, the care takers can plan on doing more intensive and drawn out tests on the patient.

# Financial Impact:
# The data shows that these "abnormal" test results will most likely have a correlation with a higher bill and longer stay in the hospital.
# If these hospitals are aware of these patterns, they can anticipate the costs and can predict which patients are more likely to have higher expenses.
# This will help with budgeting, insurance negotiation, and will allow them to plan a cost-efficient route.

# Clinical Impact:
# In the clinical world, the model can assist in decision making.
# For example, if there is a patient profile that has a big chance at returning abnormal test results, clinicians can prioritize early intervention and additional testing.
# This may lead earlier treatment with less complications.
# In addition to this, understanding which medical condition and what other causations are associated with abnormal test results will improve preventive care.

# Patient Experience Impact:
# Predictive analytics will decrease waiting time by figuring out which cases require more intense care.
# By anticipating higher-need patients, hospitals can heighten their scheduling, reduce their delays, and enhance patient satisfactory.
# This smoother care will also help calm patients and decrease the odds of false diagnoses.
